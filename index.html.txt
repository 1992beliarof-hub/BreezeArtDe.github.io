    <script>
    document.addEventListener('DOMContentLoaded', function() {
        
        // Datenstruktur für Haupt- und Unterkategorien
        const subCategoryMap = {
            'digital': ['Alle', 'Fanart', 'Grußkarten', 'Abstrakte Kunst', 'Frauen', 'Porträts', 'Kalender'],
            'epoxyd': ['Alle', 'Normal', 'Patronenhülsen', 'Leaves', 'All-in-one'],
            'epoxyd-wood': ['Alle', 'Beistelltischplatten', 'Tischplatten'],
            'photo': ['Alle', 'Landschaften']
        };

        // Hilfsfunktion, um Text in einen Code-freundlichen String umzuwandeln
        const toDataAttr = (text) => text.toLowerCase().replace(/\s+/g, '-');

        const mainTabButtons = document.querySelectorAll('.tab-btn');
        const subCategoryContainer = document.getElementById('subCategoryTabs');
        const galleryItems = document.querySelectorAll('.gallery-item');
        
        // Logik für Hauptkategorie-Filter
        mainTabButtons.forEach(button => {
            button.addEventListener('click', () => {
                mainTabButtons.forEach(btn => btn.classList.remove('active'));
                button.classList.add('active');
                
                const category = button.dataset.category;
                updateSubCategoryTabs(category);
                filterItems();
            });
        });

        // Funktion zum Erstellen der Unterkategorie-Filter
        function updateSubCategoryTabs(category) {
            subCategoryContainer.innerHTML = ''; // Leeren

            if (category !== 'alle' && subCategoryMap[category]) {
                const subCategories = subCategoryMap[category];
                subCategories.forEach(subCat => {
                    const btn = document.createElement('button');
                    btn.className = 'sub-tab-btn';
                    btn.textContent = subCat;
                    btn.dataset.subcategory = toDataAttr(subCat);
                    if (subCat === 'Alle') {
                        btn.classList.add('active');
                    }
                    btn.addEventListener('click', () => {
                        // Alle Sub-Buttons deaktivieren, dann den geklickten aktivieren
                        subCategoryContainer.querySelectorAll('.sub-tab-btn').forEach(sBtn => sBtn.classList.remove('active'));
                        btn.classList.add('active');
                        filterItems();
                    });
                    subCategoryContainer.appendChild(btn);
                });
            }
        }
        
        // Filterfunktion für beide Ebenen
        function filterItems() {
            const activeMainCategory = document.querySelector('.tab-btn.active').dataset.category;
            const activeSubCategoryBtn = document.querySelector('.sub-tab-btn.active');
            const activeSubCategory = activeSubCategoryBtn ? activeSubCategoryBtn.dataset.subcategory : 'alle';

            galleryItems.forEach(item => {
                const itemMainCategory = item.dataset.category;
                const itemSubCategory = item.dataset.subcategory;

                // Bedingungen für Sichtbarkeit
                const mainMatch = activeMainCategory === 'alle' || activeMainCategory === itemMainCategory;
                const subMatch = !activeSubCategoryBtn || activeSubCategory === 'alle' || activeSubCategory === itemSubCategory;
                
                // Ein Item ist nur sichtbar, wenn es zur Hauptkategorie passt UND (falls vorhanden) zur Unterkategorie
                if (mainMatch && subMatch) {
                    item.classList.add('visible');
                } else {
                    item.classList.remove('visible');
                }
            });
        }

        // Initialisierung
        filterItems(); // Initial alles anzeigen

        // Mobile Navigation
        const hamburger = document.getElementById('hamburger');
        const navLinks = document.getElementById('navLinks');
        hamburger.addEventListener('click', () => {
            navLinks.classList.toggle('nav-active');
        });

        // Modal-Fenster Logik
        const modal = document.getElementById('modal');
        const closeBtn = document.querySelector('.close-btn');
        const detailsButtons = document.querySelectorAll('.details-btn');

        detailsButtons.forEach(button => {
            button.addEventListener('click', () => {
                const item = button.closest('.gallery-item');
                const img = item.querySelector('img');
                const title = item.querySelector('h3').textContent;
                const description = item.querySelector('.description').textContent;
                const price = item.querySelector('.price').textContent;

                document.getElementById('modalImage').src = img.src.replace('400x250', '800x500');
                document.getElementById('modalTitle').textContent = title;
                document.getElementById('modalDescription').textContent = description;
                document.getElementById('modalPrice').textContent = price;

                modal.classList.add('active');
            });
        });

        function closeModal() {
            modal.classList.remove('active');
        }

        closeBtn.addEventListener('click', closeModal);
        modal.addEventListener('click', (e) => {
            if (e.target === modal) {
                closeModal();
            }
        });

    });
